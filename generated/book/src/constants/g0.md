Standard gravitational acceleration at Earth sea level.

## Rust Usage

```rust
use eng::{constants};
use equations::get_constant;

let c = constants::g0();
assert_eq!(c.key, "g0");
println!("g0 = g₀ {}", c.value, c.unit);

let by_id = get_constant("g0").expect("constant lookup");
assert_eq!(by_id.key, "g0");
let by_alias = get_constant("standard_gravity").expect("alias lookup");
assert_eq!(by_alias.key, "g0");
```
